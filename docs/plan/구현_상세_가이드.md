async function withOpenClawLock(
    jobLog: (msg: string) => Promise<any>,
    fn: () => Promise<any>
): Promise<any> {
    const lockDir = '/tmp/openclaw_execution.lock';
    while (true) {
        try {
            fs.mkdirSync(lockDir); // 원자적 연산
            break;
        } catch (err: any) {
            if (err.code === 'EEXIST') {
                await jobLog("Waiting for another AI process...");
                await new Promise(r => setTimeout(r, 2000));
            } else throw err;
        }
    }
    try {
        return await fn();
    } finally {
        fs.rmdirSync(lockDir, { recursive: true });
    }
}

```typescript
async function runAiReviewLoop(
    job, productCfg, skillDir, runStream, patches, isResumeMode, isLinux
): Promise<any[]> {
    const BATCH_SIZE = productCfg.aiBatchSize ?? 5;
    let finalReviewedPatches = [];

    // RAG prompt-injection
    let ragExclusions = '';
    if (productCfg.ragExclusion?.type === 'prompt-injection' ||
        productCfg.ragExclusion?.type === 'both') {
        // query_rag.py 실행 (cwd=os/linux/ 고정)
        // 결과를 ragExclusions 문자열로 조립
    }

    // file-hiding
    if (productCfg.ragExclusion?.type === 'file-hiding' ||
        productCfg.ragExclusion?.type === 'both') {
        // normalized/ → normalized_hidden/
        // patches_for_llm_review_*.json → *.hidden
    }

    for (let i = 0; i < patches.length; i += BATCH_SIZE) {
        const batch = patches.slice(i, i + BATCH_SIZE);
        cleanupSessions();  // 각 배치 전 세션 정리

        let attempt = 0;
        let parsedJson = null;
        let zodError = null;

        while (attempt <= 2) {
            attempt++;
            try {
                const prompt = productCfg.buildPrompt(skillDir, batch.length,
                    batch.map(isLinux ? prunePatchLinux : prunePatch));

                const output = await withOpenClawLock(job.log, async () => {
                    return runStream('openclaw', [
                        'agent:main', '--json', '--timeout', '1800',
                        '--session-id', `${productCfg.id}_${job.id}_batch_${i}`,
                        '-m', prompt + (zodError ? `\n\nZOD ERROR: ${zodError}` : '')
                    ]);
                });

                const extracted = extractJsonArray(output);
                const validated = ReviewSchema.parse(extracted);

                // batchValidation 체크
                if (productCfg.aiBatchValidation === 'exact' &&
                    validated.length !== batch.length) {
                    throw new Error(`Expected ${batch.length} items, got ${validated.length}`);
                }

                parsedJson = validated;
                break;
            } catch (err) {
                zodError = err.message;
                if (err.message.includes('rate limit')) {
                    fs.writeFileSync(productCfg.rateLimitFlag, '');
                    throw err;
                }
            }
        }

        if (parsedJson) {
            finalReviewedPatches.push(...parsedJson);
            // 중간 저장
            fs.writeFileSync(outputReportPath, JSON.stringify(finalReviewedPatches, null, 2));
        }
    }

    // file-hiding 복원 (finally 패턴)
    // normalizedDir_hidden → normalizedDir 복원
    // *.hidden → 원본으로 복원

    return finalReviewedPatches;
}
```

// src/app/api/pipeline/mariadb/run/route.ts (예시)
import { NextRequest, NextResponse } from 'next/server';
import { pipelineQueue } from '@/lib/queue';
import { PRODUCT_MAP } from '@/lib/products-registry';

export async function POST(req: NextRequest) {
    const { isRetry = false, isAiOnly = false } = await req.json();

    const productCfg = PRODUCT_MAP['mariadb'];
    if (!productCfg) return NextResponse.json({ error: 'Product not found' }, { status: 404 });

    // 이미 실행 중인 잡 확인
    const activeJobs = await pipelineQueue.getActive();
    const waitingJobs = await pipelineQueue.getWaiting();
    if (activeJobs.length > 0 || waitingJobs.length > 0) {
        const existingJob = activeJobs[0] || waitingJobs[0];
        return NextResponse.json({
            error: 'A pipeline job is already in the queue.',
            jobId: existingJob.id
        }, { status: 409 });
    }

    const job = await pipelineQueue.add(
        productCfg.jobName,
        { isRetry, isAiOnly }
    );

    return NextResponse.json({ success: true, jobId: job.id });
}
```

// src/app/api/pipeline/mariadb/finalize/route.ts (예시)
import Papa from 'papaparse';
import fs from 'fs';
import path from 'path';
import { prisma } from '@/lib/db';
import { getSkillDir, PRODUCT_MAP } from '@/lib/products-registry';

export async function POST() {
    const productCfg = PRODUCT_MAP['mariadb'];
    const skillDir = getSkillDir(productCfg);

    const patches = await prisma.reviewedPatch.findMany({
        where: { vendor: productCfg.vendorString },
        orderBy: { reviewedAt: 'desc' }
    });

    const rows = patches.map(p => ({
        IssueID: p.issueId,
        Component: p.component,
        // ...
    }));

    let csvContent = Papa.unparse(rows);
    if (productCfg.csvBOM) csvContent = '\uFEFF' + csvContent;

    const csvPath = path.join(skillDir, productCfg.finalCsvFile);
    fs.writeFileSync(csvPath, csvContent, 'utf-8');

}

### 6.1 신규 제품 SKILL.md 최소 구조

```markdown
# {Product} Patch Evaluation Rules

## 1. Overview

This SKILL guides AI evaluation of {Product} security patches for enterprise IT infrastructure.

## 2. Input Format

Each entry in [BATCH DATA] contains:
- patch_id: Unique patch identifier
- description: Patch description
- component: Affected component
- version: Patch version
- issued_date: Release date (YYYY-MM-DD)

## 3. Evaluation Criteria

### 3.1 Include Criteria (Decision: Done)
- CVSSv3 score ≥ 7.0 (High or Critical)
- Actively exploited vulnerability (CISA KEV listed)
- Remote Code Execution or Privilege Escalation
- Data integrity or confidentiality impact

### 3.2 Exclude Criteria (Decision: Exclude)
- CVSSv3 score < 7.0 (Moderate, Low)
- Development-only tools with no production impact
- Duplicate advisories

### 3.3 Review Date Window (CRITICAL)

**Non-kernel components:**
- Review only patches issued 90~180 days ago
- Patches issued within last 90 days: EXCLUDE (immature, may be superseded)

**Kernel components** (kernel, kernel-uek, linux-image-*):
- Review patches issued 0~180 days ago (include recent kernel patches)

## 4. Output Format

### 4.1 JSON Schema

Return a pure JSON array (NO markdown fences). Each object:

```json
{
  "IssueID": "string",
  "Component": "string",
  "Version": "string",
  "Vendor": "{Vendor}",
  "Date": "YYYY-MM-DD",
  "Criticality": "Critical|Important|Moderate|Low",
  "Description": "1-2 sentence English summary",
  "KoreanDescription": "한국어 1-2문장 요약",
  "Decision": "Done|Exclude",
  "Reason": "string (Exclude 시 사유)"
}
```

### 4.2 Criticality Mapping

| CVSS Score | Criticality |
|------------|-------------|
| 9.0 - 10.0 | Critical |
| 7.0 - 8.9  | Important |
| 4.0 - 6.9  | Moderate |
| 0.0 - 3.9  | Low |

### 4.3 Output Validation Rules

1. Return EXACTLY {batchSize} objects (one per input patch)
2. IssueID MUST match the input patch_id exactly
3. Vendor MUST be "{Vendor}" for all items
4. Date MUST be YYYY-MM-DD format
5. NO null values — use "Unknown" if data unavailable

## 5. General Rules

1. NEVER use tools to read files in the workspace
2. NEVER recall past RAG memories or previous reviews
3. Base evaluation SOLELY on [BATCH DATA] provided
4. Korean descriptions must be natural, not machine-translated
5. Hallucination prevention: If patch data is ambiguous, default to Decision='Pending'
```

### 6.2 전처리 스크립트 기본 구조

```python
#!/usr/bin/env python3
"""
{product}_preprocessing.py — {Product} 패치 전처리 스크립트

사용법:
    python3 {product}_preprocessing.py --days 180
"""
import argparse
import json
import os
from datetime import datetime, timedelta
from pathlib import Path

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument('--days', type=int, default=180)
    args = parser.parse_args()

    script_dir = Path(__file__).parent
    data_dir = script_dir / '{product}_data'

    cutoff_date = datetime.now() - timedelta(days=args.days)

    patches = []
    for json_file in data_dir.glob('*.json'):
        with open(json_file) as f:
            raw = json.load(f)

        # 날짜 필터링
        issued = datetime.fromisoformat(raw.get('date', '1970-01-01'))
        if issued < cutoff_date:
            continue

        # 정규화
        patch = {
            'patch_id': raw.get('id'),
            'vendor': '{Product}',
            'component': raw.get('component', '{product}'),
            'version': raw.get('version', ''),
            'os_version': raw.get('os_version'),
            'description': (raw.get('description', ''))[:4000],
            'issued_date': issued.strftime('%Y-%m-%d')
        }
        patches.append(patch)

    output_file = script_dir / 'patches_for_llm_review_{product}.json'
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(patches, f, ensure_ascii=False, indent=2)

    # 반드시 이 태그 출력 (대시보드 카운트 파싱용)
    print(f'[{LOGTAG}-PREPROCESS_DONE] count={len(patches)}')

if __name__ == '__main__':
    main()
```

## 9. 자주 발생하는 오류 및 해결

| 오류 증상 | 원인 | 해결 |
|----------|------|------|
| 빌드 실패: `Promise<void>` 타입 | `job.log()` 반환 타입 불일치 | `Promise<any>`로 변경 |
| AI 리뷰 결과가 0 | `preprocessedPatchMapper` 필드명 오류 | 실제 JSON 필드 확인 |
| 파이프라인이 Linux 경로로 폴스루 | `job.name` 분기 누락 | queue.ts에 `run-{product}-pipeline` 브랜치 추가 |
| `gateway closed` 오류 | SKILL.md 없음 또는 100줄 미만 | SKILL.md 생성/보완 |
| 세션 오염으로 AI 환각 | `sessions.json` 미삭제 | `cleanupSessions()`에 `sessions.json` 삭제 추가 |
| file-hiding 복원 실패 | `finally` 블록 누락 | try-finally 패턴으로 복원 보장 |
| Turbopack 빌드 패닉 | venv symlink | `patch-review/` 내 venv 삭제 |
| `collected` 카운트 0 | rawDataFilePrefix 오류 | `ls {product}_data/ | head` 로 실제 prefix 확인 |
| `getIncompleteProducts()` 오작동 | finalCsvFile 경로 오류 | `getSkillDir(p)`로 절대경로 계산 확인 |
| CSV 한글 깨짐 (Excel) | BOM 누락 | `csvBOM: true` 설정 |
